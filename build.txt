name:
	giti
build:
  &dev:WorksOfBarry
  
  CRTSRCPF FILE(&INTOLIB/QSOURCE) RCDLEN(112)
  CPYFRMSTMF FROMSTMF('&DIR/QSOURCE/giti.cmd') TOMBR('/QSYS.lib/&INTOLIB.lib/QSOURCE.file/GITI.mbr') MBROPT(*ADD)
  CPYFRMSTMF FROMSTMF('&DIR/QSOURCE/screen.dspf') TOMBR('/QSYS.lib/&INTOLIB.lib/QSOURCE.file/SCREEN.mbr') MBROPT(*ADD)
  
  //Create display file
  CRTDSPF FILE(&INTOLIB/SCREEN) SRCFILE(&INTOLIB/QSOURCE) SRCMBR(SCREEN) REPLACE(*YES)
  
  //Create programs
  CRTSQLRPGI OBJ(&INTOLIB/GITLOGPRSE) SRCSTMF('&DIR/QSOURCE/gitlogprse.sqlrpgle') COMMIT(*NONE) REPLACE(*YES) DBGVIEW(*SOURCE)
  CRTSQLRPGI OBJ(&INTOLIB/GITLOGPRSE) SRCSTMF('&DIR/QSOURCE/main.sqlrpgle') COMMIT(*NONE) REPLACE(*YES) DBGVIEW(*SOURCE)
  
  //Create command
  CRTCMD CMD(&INTOLIB/GITI) PGM(&INTOLIB/MAIN) SRCFILE(&INTOLIB/QSOURCE) SRCMBR(GITI) REPLACE(*YES) TEXT('Launch giti application')
